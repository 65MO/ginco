143 	/*
144 	 * (non-Javadoc)
145 	 * 
146 	 * @see
147 	 * fr.mcc.ginco.imports.ISKOSImportService#importSKOSFile(java.lang.String,
148 	 * java.lang.String, java.io.File)
149 	 */
150 	@Override
151 	public Thesaurus importSKOSFile(String fileContent, String fileName,
152 			File tempDir) throws BusinessException {
153 		URI fileURI = writeTempFile(fileContent, fileName, tempDir);
154 		Thesaurus thesaurus = null;
155 		try {
156 			// Reader init
157 			Model model = ModelFactory.createDefaultModel();
158 			InputStream in = FileManager.get().open(fileURI.toString());
159 			model.read(in, null);
160 
161 			// Getting thesaurus
162 			Resource thesaurusSKOS = getSKOSThesaurus(model);
163 			if (thesaurusSKOS == null) {
164 				logger.error("no thesaurus found");
165 			} else {
166 				if (thesaurusDAO.getById(thesaurusSKOS.getURI()) != null) {
167 					throw new BusinessException(
168 							"Trying to import an existing thesaurus "
169 									+ thesaurusSKOS.getURI(),
170 							"import-already-existing-thesaurus");
171 				}
172 				thesaurus = thesaurusBuilder.buildThesaurus(thesaurusSKOS,
173 						model);
174 				thesaurusDAO.update(thesaurus);
175 			}
176 
177 			List<Resource> skosConcepts = getSKOSRessources(model, SKOS.CONCEPT);
178 			buildConcepts(thesaurus, skosConcepts);
179 			buildConceptsAssociations(thesaurus, skosConcepts);
180 			buildConceptsRoot(thesaurus, skosConcepts);
181 			buildArrays(thesaurus, model);
182 
183 		} catch (JenaException je) {
184 			throw new BusinessException("Error reading imported file ",
185 					"import-unable-to-read-file", je);
186 		} finally {
187 			deleteTempFile(fileURI);
188 		}
189 		return thesaurus;
190 	}

193 	 * Builds the thesaurus arrays from the model
194 	 * 
195 	 * @param thesaurus
196 	 * @param model
197 	 */
198 	private void buildArrays(Thesaurus thesaurus, Model model) {
199 		List<Resource> skosCollections = getSKOSRessources(model,
200 				SKOS.COLLECTION);
201 		for (Resource skosCollection : skosCollections) {
202 			ThesaurusArray array = arrayBuilder.buildArray(skosCollection,
203 					model, thesaurus);
204 			thesaurusArrayDAO.update(array);
205 
206 			Statement stmtLabel = skosCollection.getProperty(SKOS.PREF_LABEL);
207 			NodeLabel nodeLabel = nodeLabelBuilder.buildNodeLabel(stmtLabel,
208 					model, thesaurus, array);
209 			nodeLabelDAO.update(nodeLabel);
210 		}
211 	}

252 	/**
253 	 * Builds the concept with minimal informations and it's terms and notes
254 	 * 
255 	 * @param thesaurus
256 	 * @param skosConcepts
257 	 */
258 	private void buildConcepts(Thesaurus thesaurus, List<Resource> skosConcepts) {
259 		for (Resource skosConcept : skosConcepts) {
260 			// Minimal concept informations
261 			ThesaurusConcept concept = conceptBuilder.buildConcept(skosConcept,
262 					thesaurus);
263 			thesaurusConceptDAO.update(concept);
264 
265 			// Concept terms
266 			List<ThesaurusTerm> terms = termBuilder.buildTerms(skosConcept,
267 					thesaurus, concept);
268 			for (ThesaurusTerm term : terms) {
269 				thesaurusTermDAO.update(term);
270 			}
271 
272 			// Concept notes
273 			List<Note> conceptNotes = conceptNoteBuilder.buildConceptNotes(
274 					skosConcept, concept, thesaurus);
275 			for (Note conceptNote : conceptNotes) {
276 				noteDAO.update(conceptNote);
277 			}
278 		}
279 	}

281 	/**
282 	 * Gets the thesaurus resource from the model, returning the first one only
283 	 * 
284 	 * @param model
285 	 * @return
286 	 */
287 	private Resource getSKOSThesaurus(Model model) {
288 		SimpleSelector schemeSelector = new SimpleSelector(null, null,
289 				(RDFNode) null) {
290 			public boolean selects(Statement s) {
291 				if (s.getObject().isResource()) {
292 					return s.getObject().asResource()
293 							.equals(SKOS.CONCEPTSCHEME);
294 				} else {
295 					return false;
296 				}
297 			}
298 		};
299 
300 		StmtIterator iter = model.listStatements(schemeSelector);
301 
302 		Resource thesaurusSKOS = null;
303 		if (iter.hasNext()) {
304 			Statement s = iter.next();
305 			thesaurusSKOS = s.getSubject().asResource();
306 		}
307 		return thesaurusSKOS;
308 	}

310 	/**
311 	 * Gets the list of resources from the gien model
312 	 * 
313 	 * @param model
314 	 * @param resource
315 	 * @return
316 	 */
317 	private List<Resource> getSKOSRessources(Model model,
318 			final Resource resource) {
319 		SimpleSelector schemeSelector = new SimpleSelector(null, null,
320 				(RDFNode) null) {
321 			public boolean selects(Statement s) {
322 				if (s.getObject().isResource()) {
323 					return s.getObject().asResource().equals(resource);
324 				} else {
325 					return false;
326 				}
327 			}
328 		};
329 		StmtIterator iter = model.listStatements(schemeSelector);
330 
331 		List<Resource> skosRessources = new ArrayList<Resource>();
332 		while (iter.hasNext()) {
333 			Statement s = iter.next();
334 			skosRessources.add(s.getSubject().asResource());
335 		}
336 		return skosRessources;
337 	}

344 	private URI writeTempFile(String fileContent, String fileName, File tempDir)
345 			throws BusinessException {
346 		logger.debug("Writing temporary file for import");
347 		String prefix = fileName.substring(0, fileName.lastIndexOf("."));
348 		logger.debug("Filename : " + prefix);
349 		File file;
350 		try {
351 			file = File.createTempFile(prefix, ".tmp", tempDir);
352 
353 			FileWriter fileWriter = new FileWriter(file);
354 			fileWriter.write(fileContent);
355 			fileWriter.close();
356 		} catch (IOException e) {
357 			throw new BusinessException(
358 					"Error storing temporarty file for import " + prefix,
359 					"import-unable-to-write-temporary-file", e);
360 		}
361 		return file.toURI();
362 	}
