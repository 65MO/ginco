189 	public List<ThesaurusConcept> getRootConcepts(ThesaurusConcept concept) {
190 		ThesaurusConcept start;
191 		HashMap<String, Integer> path = new HashMap<String, Integer>();
192 		Set<ThesaurusConcept> roots = new HashSet<ThesaurusConcept>();
193 		path.clear();
194 		roots.clear();
195 		path.put(concept.getIdentifier(), 0);
196 		start = concept;
197 		getRoot(concept, 0, start, path, roots);
198 		return new ArrayList<ThesaurusConcept>(roots);
199 	}

203 			List<String> parentsToRemove) throws BusinessException {

219 	private void getRoot(ThesaurusConcept concept, Integer iteration,
220 			ThesaurusConcept start, Map<String, Integer> path,
221 			Set<ThesaurusConcept> roots) {
222 		iteration++;
223 		Set<ThesaurusConcept> directParents = concept.getParentConcepts();
224 		if (directParents.isEmpty()) {
225 			if (iteration != 1) {
226 				roots.add(concept);
227 			}
228 			return;
229 		}
230 		boolean flag = false;
231 		Set<ThesaurusConcept> stack = new HashSet<ThesaurusConcept>();
232 		for (ThesaurusConcept directParent : directParents) {
233 			if (path.containsKey(directParent.getIdentifier())) {
234 				continue;
235 			} else {
236 				path.put(directParent.getIdentifier(), iteration);
237 				stack.add(directParent);
238 				flag = true;
239 			}
240 		}
241 
242 		// HACK to deal with cyclic dependencies. Should be reThink in some
243 		// time...
244 		if (!flag && directParents.size() == 1 && directParents.contains(start)) {
245 			roots.add(concept);
246 		}
247 
248 		if (!stack.isEmpty()) {
249 			for (ThesaurusConcept toVisit : stack) {
250 				getRoot(toVisit, iteration, start, path, roots);
251 			}
252 			stack.clear();
253 		}

279 	@Override
280 	public ThesaurusConcept updateThesaurusConcept(ThesaurusConcept object,
281 			List<ThesaurusTerm> terms, List<String> associatedConceptIds)
282 			throws BusinessException {
283 
284 		if (object.getStatus() == ConceptStatusEnum.CANDIDATE.getStatus()) {
285 			// We can set status = candidate only if concept has not relation
286 			// (both hierarchical or associative)
287 			if (!associatedConceptIds.isEmpty()
288 					// || !object.getAssociativeRelationshipRight().isEmpty()
289 					|| !object.getParentConcepts().isEmpty()
290 					|| hasChildren(object.getIdentifier())) {
291 				throw new BusinessException(
292 						"A concept must not have a status candidate when it still have relations",
293 						"concept-status-candidate-relation");
294 			}
295 		}
296 
297 		if (object.getStatus() == ConceptStatusEnum.VALIDATED.getStatus()) {
298 			// Test if parent concepts are validated
299 			Set<ThesaurusConcept> parents = object.getParentConcepts();
300 			for (ThesaurusConcept parent : parents) {
301 				if (parent.getStatus() != ConceptStatusEnum.VALIDATED
302 						.getStatus()) {
303 					throw new BusinessException(
304 							"A concept cannot have a parent which status is not validated",
305 							"concept-parent-not-validated");
306 				}
307 			}
308 
309 
310 		}
311 		object = saveAssociativeRelationship(object, associatedConceptIds);
312 
313 		ThesaurusConcept concept = thesaurusConceptDAO.update(object);
314 		updateConceptTerms(concept, terms);
315 		//indexerService.addConcept(concept);
316 		return concept;
317 	}

319 	private ThesaurusConcept saveAssociativeRelationship(
320 			ThesaurusConcept concept, List<String> associatedConceptIds)
321 			throws BusinessException {
322 		Set<AssociativeRelationship> relations = new HashSet<AssociativeRelationship>();
323 		if (concept.getAssociativeRelationshipLeft() == null) {
324 			concept.setAssociativeRelationshipLeft(new HashSet<AssociativeRelationship>());
325 		} else {
326 			for (AssociativeRelationship associativeRelationship: concept.getAssociativeRelationshipLeft()) {
327 				associativeRelationshipDAO.delete(associativeRelationship);
328 			}
329 		}
330 		concept.getAssociativeRelationshipLeft().clear();
331 
332 		if (concept.getAssociativeRelationshipRight() == null) {
333 			concept.setAssociativeRelationshipRight(new HashSet<AssociativeRelationship>());
334 		}else {
335 			for (AssociativeRelationship associativeRelationship: concept.getAssociativeRelationshipRight()) {
336 				associativeRelationshipDAO.delete(associativeRelationship);
337 			}
338 		}
339 		concept.getAssociativeRelationshipRight().clear();
340 
341 		for (String associatedConceptsId : associatedConceptIds) {
342 			logger.debug("Settings associated concept " + associatedConceptsId);
343 			ThesaurusConcept linkedThesaurusConcept = getThesaurusConceptById(associatedConceptsId);
344 			if (linkedThesaurusConcept.getStatus() != ConceptStatusEnum.VALIDATED
345 					.getStatus()) {
346 				throw new BusinessException(
347 						"A concept must associate a validated concept",
348 						"concept-associate-validated-concept");
349 			}
350 			List<String> alreadyAssociatedConcepts = associativeRelationshipDAO
351 					.getAssociatedConcepts(linkedThesaurusConcept);
352 			if (!alreadyAssociatedConcepts.contains(concept.getIdentifier())) {
353 				AssociativeRelationship relationship = new AssociativeRelationship();
354 				AssociativeRelationship.Id relationshipId = new AssociativeRelationship.Id();
355 				relationshipId.setConcept1(concept.getIdentifier());
356 				relationshipId.setConcept2(associatedConceptsId);
357 				relationship.setIdentifier(relationshipId);
358 				relationship.setConceptLeft(concept);
359 				relationship.setConceptRight(thesaurusConceptDAO
360 						.getById(associatedConceptsId));
361 				relationship
362 						.setRelationshipRole(associativeRelationshipRoleDAO.getDefaultAssociativeRelationshipRole()	);
363 				relations.add(relationship);
364 				associativeRelationshipDAO.update(relationship);
365 			}
366 		}
367 		concept.getAssociativeRelationshipLeft().addAll(relations);
368 
369 		return concept;
370 	}

374 	public ThesaurusConcept destroyThesaurusConcept(ThesaurusConcept object)
375 			throws BusinessException {
376 		List<ThesaurusTerm> terms = thesaurusTermDAO
377 				.findTermsByConceptId(object.getIdentifier());
378 		for (ThesaurusTerm term : terms) {
379 			term.setConcept(null);
380 			thesaurusTermDAO.update(term);
381 		}
382 
383 		List<ThesaurusConcept> childrenConcepts = getChildrenByConceptId(object
384 				.getIdentifier());
385 		for (ThesaurusConcept childConcept : childrenConcepts) {
386 			childConcept.getParentConcepts().remove(object);
387 			thesaurusConceptDAO.update(childConcept);
388 		}
389 
390 		List<ThesaurusConcept> rootChildrenConcepts = thesaurusConceptDAO
391 				.getAllRootChildren(object);
392 		for (ThesaurusConcept rootChild : rootChildrenConcepts) {
393 			rootChild.getRootConcepts().remove(object);
394 			thesaurusConceptDAO.update(rootChild);
395 		}
396 
397 		List<ThesaurusArray> arrays = thesaurusArrayDAO
398 				.getConceptSuperOrdinateArrays(object.getIdentifier());
399 		for (ThesaurusArray array : arrays) {
400 			thesaurusArrayDAO.delete(array);
401 		}
402 
403 		return thesaurusConceptDAO.delete(object);
404 	}

442 	private void calculateChildrenRoots(String parentId, String originalParentId) {
443 		List<ThesaurusConcept> childrenConcepts = getChildrenByConceptId(parentId);
444 		for (ThesaurusConcept concept : childrenConcepts) {
445 			if (concept.getIdentifier() != originalParentId) {
446 				logger.info("calculating root concept for chiled with concept Id : "
447 						+ concept.getIdentifier());
448 				List<ThesaurusConcept> thisChildRoots = getRootConcepts(concept);
449 				concept.setRootConcepts(new HashSet<ThesaurusConcept>(
450 						thisChildRoots));
451 				calculateChildrenRoots(concept.getIdentifier(),
452 						originalParentId);
453 			}
454 		}
455 	}

490     @Override
491     public int getConceptsHierarchicalRelations(String firstConceptId, String secondConceptId) 
492     		throws BusinessException{
493     	ThesaurusConcept firstConcept = thesaurusConceptDAO.getById(firstConceptId);
494     	ThesaurusConcept secondConcept = thesaurusConceptDAO.getById(secondConceptId);
495     	
496     	if (firstConcept != null && secondConcept != null){
497     		if (thesaurusConceptDAO.getChildrenConcepts(firstConceptId)
498         			.contains(secondConcept)){
499         		return ConceptHierarchicalRelationsEnum.PARENT.getStatus();
500         	}
501         	
502         	if (thesaurusConceptDAO.getChildrenConcepts(secondConceptId)
503         			.contains(firstConcept)){
504         		return ConceptHierarchicalRelationsEnum.CHILD.getStatus();
505         	}
506         	
507         	else   
508         		return ConceptHierarchicalRelationsEnum.NORELATIONS.getStatus();
509     	}
510     	else{
511     		throw new BusinessException("One or both concepts don't exist", "concepts-do-not-exist");
512     	}
