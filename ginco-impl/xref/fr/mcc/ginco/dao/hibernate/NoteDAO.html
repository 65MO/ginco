35  package fr.mcc.ginco.dao.hibernate;

37  import fr.mcc.ginco.beans.Note;
38  import fr.mcc.ginco.dao.INoteDAO;
39  import org.hibernate.criterion.Order;
40  import org.hibernate.criterion.Projections;
41  import org.hibernate.criterion.Restrictions;
42  import org.springframework.stereotype.Repository;
43  
44  import java.util.List;
45  
46  /**
47   * Implementation of {@link INoteDAO}; basic class for DAO-related work.
48   */
49  @Repository("noteDAO")
50  public class NoteDAO extends GenericHibernateDAO<Note, String>
51  		implements INoteDAO {
52  
53  	public NoteDAO() {
54  		super(Note.class);
55  	}
56  
57  	/* (non-Javadoc)
58  	 * @see fr.mcc.ginco.dao.INoteDAO#findConceptNotes(java.lang.String)
59  	 */
60  	@Override
61  	public List<Note> findConceptPaginatedNotes(String conceptId, Integer startIndex, Integer limit) {
62  		return getCurrentSession().createCriteria(Note.class)
63  				.setMaxResults(limit)
64  				.add(Restrictions.eq("concept.identifier",conceptId))
65  				.setFirstResult(startIndex).addOrder(Order.asc("lexicalValue"))
66  				.list();
67  	}
68  
69  	/* (non-Javadoc)
70  	 * @see fr.mcc.ginco.dao.INoteDAO#findTermNotes(java.lang.String)
71  	 */
72  	@Override
73  	public List<Note> findTermPaginatedNotes(String termId, Integer startIndex, Integer limit) {
74  		return getCurrentSession().createCriteria(Note.class)
75  				.setMaxResults(limit)
76  				.add(Restrictions.eq("term.identifier",termId))
77  				.setFirstResult(startIndex).addOrder(Order.asc("lexicalValue"))
78  				.list();
79  		}
80  
81  	/* (non-Javadoc)
82  	 * @see fr.mcc.ginco.dao.INoteDAO#getConceptNoteCount(java.lang.String)
83  	 */
84  	@Override
85  	public Long getConceptNoteCount(String conceptId) {
86  		return (Long) getCurrentSession()
87  				.createCriteria(Note.class)
88  				.add(Restrictions.eq("concept.identifier",conceptId))
89  				.setProjection(Projections.rowCount())
90  				.list().get(0);
91  	}
92  
93  	/* (non-Javadoc)
94  	 * @see fr.mcc.ginco.dao.INoteDAO#getTermNoteCount(java.lang.String)
95  	 */
96  	@Override
97  	public Long getTermNoteCount(String termId) {
98  		return (Long) getCurrentSession()
99  				.createCriteria(Note.class)
100 				.add(Restrictions.eq("term.identifier",termId))
101 				.setProjection(Projections.rowCount())
102 				.list().get(0);
103 	}
