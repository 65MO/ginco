90  	private ThesaurusTerm buildTerm(Statement stmt,
91  			Thesaurus thesaurus, ThesaurusConcept concept, boolean preferred, boolean hidden)
92  			throws BusinessException {
93  		logger.debug("building term " + stmt.getString());
94  		ThesaurusTerm term = new ThesaurusTerm();
95  		term.setConcept(concept);
96  		term.setCreated(thesaurus.getCreated());
97  		term.setIdentifier(generatorService.generate(ThesaurusTerm.class));
98  		term.setLexicalValue(stmt.getString().trim());
99  		term.setModified(thesaurus.getDate());
100 		term.setPrefered(preferred);
101         term.setHidden(hidden);
102 		term.setRole(thesaurusTermRoleDAO.getDefaultThesaurusTermRole());
103 		term.setStatus(TermStatusEnum.VALIDATED.getStatus());
104 		term.setThesaurus(thesaurus);
105 
106 		RDFNode prefLabel = stmt.getObject();
107 		String lang = prefLabel.asLiteral().getLanguage();
108 		if (StringUtils.isEmpty(lang)) {
109 			Language defaultLangL = languagesDAO.getById(defaultLang);
110 			term.setLanguage(defaultLangL);
111 		} else {
112 			List<Language> langs = languagesDAO.getByPart1(lang);
113 			if (langs.size() > 0) {
114 				term.setLanguage(langs.get(0));
115 			} else {
116 				throw new BusinessException("Term " + stmt.getString()
117 						+ " is missing it's language",
118 						"import-term-with-no-lang");
119 			}
120 		}
121 		return term;
122 	}

124 	/**
125 	 * Builds the list of ThesaurusTerm for a given SKOS resource concept
126 	 * @param skosConcept
127 	 * @param thesaurus
128 	 * @param concept
129 	 * @return
130 	 * @throws BusinessException
131 	 */
132 	public List<ThesaurusTerm> buildTerms(Resource skosConcept,
133 			Thesaurus thesaurus, ThesaurusConcept concept)
134 			throws BusinessException {
135 		
136 		logger.debug("building terms for concept " + concept.getIdentifier());
137 
138 		List<ThesaurusTerm> terms = new ArrayList<ThesaurusTerm>();
139 
140 		// Preferred term
141 		Statement stmtPreferred = skosConcept.getProperty(SKOS.PREF_LABEL);
142 		terms.add(buildTerm(stmtPreferred,  thesaurus, concept, true, false));
143 
144 		// Alt terms
145 		StmtIterator stmtAltItr = skosConcept.listProperties(SKOS.ALT_LABEL);
146 		while (stmtAltItr.hasNext()) {
147 			Statement stmtAlt = stmtAltItr.next();
148 			terms.add(buildTerm(stmtAlt,  thesaurus, concept, false, false));
149 
150 		}
151 		// Hidden terms
152 		StmtIterator stmtHiddenItr = skosConcept
153 				.listProperties(SKOS.HIDDEN_LABEL);
154 		while (stmtHiddenItr.hasNext()) {
155 			Statement stmtHidden = stmtHiddenItr.next();
156 			terms.add(buildTerm(stmtHidden,  thesaurus, concept, false, true));
157 		}
158 
159 		return terms;
160 	}
