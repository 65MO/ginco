318 
319 		final ThesaurusConcept node1 = new ThesaurusConcept();
320 		node1.setIdentifier("node1");
321 		node1.setThesaurus(thesaurus);
322 		final ThesaurusConcept node2 = new ThesaurusConcept();
323 		node2.setIdentifier("node2");
324 		node2.setThesaurus(thesaurus);
325 		final ThesaurusConcept node3 = new ThesaurusConcept();
326 		node3.setIdentifier("node3");
327 		node3.setThesaurus(thesaurus);
328 
329 		when(thesaurusConceptDAO.getById("node2")).thenReturn(node2);
330 
331 		node1.getParentConcepts().addAll(new ArrayList<ThesaurusConcept>() {
332 			{
333 				add(node2);
334 			}
335 		});
336 		node2.getParentConcepts().addAll(new ArrayList<ThesaurusConcept>() {
337 			{
338 				add(node3);
339 			}
340 		});
341 
342 		List<String> toRemove = new ArrayList<String>();
343 		toRemove.add(node2.getIdentifier());
344 
345 		thesaurusConceptService.removeParents(node1, toRemove);
346

348 				.getParentConcepts().size());
349 		Assert.assertTrue("Term now should be TOP!", node1.getTopConcept());
350 
351 		node1.getParentConcepts().addAll(new ArrayList<ThesaurusConcept>() {
352 			{
353 				add(node2);
354 			}
355 		});
356 		thesaurus.setDefaultTopConcept(false);
357 
358 		thesaurusConceptService.removeParents(node1, toRemove);
359 
360 		Assert.assertEquals("There should be no parents!", 0, node1
361 				.getParentConcepts().size());
362 		Assert.assertFalse("Term now should NOT be TOP!", node1.getTopConcept());
