161 		String thesaurusId = "http://www.culturecommunication.gouv.fr/th1";
162 		Thesaurus th = new Thesaurus();
163 		th.setIdentifier(thesaurusId);
164 
165 		List<ThesaurusConcept> list = thesaurusConceptDAO.getRootConcepts(
166 				th.getIdentifier(), false);
167 		Assert.assertEquals(1, list.size());
168 		Assert.assertEquals("http://www.culturecommunication.gouv.fr/co2", list
169 				.get(0).getIdentifier());
170 
171 		List<ThesaurusConcept> listWithOrphans = thesaurusConceptDAO
172 				.getRootConcepts(th.getIdentifier(), null);
173 		Assert.assertEquals(3, listWithOrphans.size());
174 		List<String> expectedOrphans = new ArrayList<String>();
175 		expectedOrphans.add("http://www.culturecommunication.gouv.fr/co1");
176 		expectedOrphans.add("http://www.culturecommunication.gouv.fr/co2");
177 		expectedOrphans.add("http://www.culturecommunication.gouv.fr/co3");
178 
179 		List<String> actualOrphansIds = new ArrayList<String>();
180 		for (ThesaurusConcept concept : listWithOrphans) {
181 			actualOrphansIds.add(concept.getIdentifier());
182 		}
183 		ListAssert.assertEquals(expectedOrphans, actualOrphansIds);
184 		List<ThesaurusConcept> listOnlyOrphans = thesaurusConceptDAO
185 				.getRootConcepts(th.getIdentifier(), true);
186 		Assert.assertEquals(2, listOnlyOrphans.size());
187 		List<String> expectedOnlyOrphans = new ArrayList<String>();
188 		expectedOnlyOrphans.add("http://www.culturecommunication.gouv.fr/co1");
189 		expectedOnlyOrphans.add("http://www.culturecommunication.gouv.fr/co3");
190 
191 		List<String> actualOnlyOrphansIds = new ArrayList<String>();
192 		for (ThesaurusConcept concept : listOnlyOrphans) {
193 			actualOnlyOrphansIds.add(concept.getIdentifier());
194 		}
195 		ListAssert.assertEquals(expectedOnlyOrphans, actualOnlyOrphansIds);
196 	}
