97  	@Override
98  	public String getThesaurusExport(Thesaurus thesaurus)
99  			throws BusinessException, JAXBException {
100 		MCCExportedThesaurus thesaurusToExport = new MCCExportedThesaurus();
101 		String thesaurusId = thesaurus.getIdentifier();
102 		
103 		//---Exporting the thesaurus
104 		thesaurusToExport.setThesaurus(thesaurus);
105 		
106 		//Exporting the thesaurus' versions
107 		thesaurusToExport.setThesaurusVersions(thesaurusVersionHistoryService.getVersionsByThesaurusId(thesaurusId));
108 		
109 		//Exporting sandboxed terms
110 		List<ThesaurusTerm> sandboxedTerms = thesaurusTermService.getPaginatedThesaurusSandoxedTermsList(0, thesaurusTermService.getSandboxedTermsCount(thesaurusId).intValue(), thesaurusId);
111 		for (ThesaurusTerm thesaurusTerm : sandboxedTerms) {
112 			thesaurusToExport.getTerms().add(thesaurusTerm);
113 		}
114 		
115 		//---Exporting the terms and the concepts of the thesaurus
116 		List<ThesaurusConcept> concepts = thesaurusConceptService.getConceptsByThesaurusId(null, thesaurusId, true, false);
117 		for (ThesaurusConcept thesaurusConcept : concepts) {
118 			
119 			//Exporting terms of the concepts
120 			List<ThesaurusTerm> terms = thesaurusTermService.getTermsByConceptId(thesaurusConcept.getIdentifier());
121 			for (ThesaurusTerm thesaurusTerm : terms) {
122 				thesaurusToExport.getTerms().add(thesaurusTerm);
123 			}
124 			
125 			//Exporting term notes
126 			for (ThesaurusTerm thesaurusTerm : terms) {
127 				JaxbList<Note> termNotes = mccTermExporter.getExportTermNotes(thesaurusTerm);
128 				if (termNotes != null && !termNotes.isEmpty()) {
129 					thesaurusToExport.getTermNotes().put(thesaurusTerm.getIdentifier(), termNotes);					
130 				}
131 			}
132 			
133 			//Exporting concept notes
134 			JaxbList<Note> conceptNotes = mccConceptExporter.getExportConceptNotes(thesaurusConcept);
135 			if (conceptNotes != null && !conceptNotes.isEmpty()) {
136 				thesaurusToExport.getConceptNotes().put(thesaurusConcept.getIdentifier(), conceptNotes);
137 			}
138 			
139 			//Exporting the parent concepts
140 			JaxbList<String> parentIds = mccConceptExporter.getExportHierarchicalConcepts(thesaurusConcept);
141 			if (parentIds != null && !parentIds.isEmpty()) {
142 				thesaurusToExport.getHierarchicalRelationship().put(thesaurusConcept.getIdentifier(), parentIds);
143 			}
144 			//Exporting relative relationship
145 			JaxbList<String> associations = mccConceptExporter.getExportAssociativeRelationShip(thesaurusConcept);
146 			if (associations != null && !associations.isEmpty()) {
147 				thesaurusToExport.getAssociativeRelationship().put(thesaurusConcept.getIdentifier(), associations);
148 			}
149 		}
150 		
151 		//---Exporting the concepts arrays of the thesaurus
152 		List<ThesaurusArray> arrays = thesaurusArrayService.getAllThesaurusArrayByThesaurusId(thesaurusId);
153 		for (ThesaurusArray thesaurusArray : arrays) {
154 			thesaurusToExport.getConceptsArrays().add(thesaurusArray);
155 		}
156 		
157 		//---Exporting the concepts groups of the thesaurus
158 		List<ThesaurusConceptGroup> groups = thesaurusConceptGroupService.getAllThesaurusConceptGroupsByThesaurusId(thesaurusId);
159 		for (ThesaurusConceptGroup thesaurusGroup : groups) {
160 			thesaurusToExport.getConceptsGroups().add(thesaurusGroup);
161 		}
162 		
163 		//We encode it in XML
164 		return serializeToXmlWithJaxb(thesaurusToExport);
165 	}
166 	
167 	/**
168 	 * This method serialize in XML a Thesaurus with all its elements (terms, concepts, relationships, notes, etc.).
169 	 * @param thesaurusToExport
170 	 * @return
171 	 * @throws JAXBException
172 	 */
173 	private String serializeToXmlWithJaxb(MCCExportedThesaurus thesaurusToExport)
174 			throws JAXBException {
175 		//This method encodes a MCCExportedThesaurus in XML
176 		String result = null;
177 		JAXBContext context = JAXBContext
178 				.newInstance(MCCExportedThesaurus.class);
179 		Marshaller marshaller = context.createMarshaller();
180 		marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, true);
181 		ByteArrayOutputStream output = new ByteArrayOutputStream();
182 		marshaller.marshal(thesaurusToExport, output);
183 		result = output.toString();
184 		return result;
185 	}
