35  package fr.mcc.ginco.dao.hibernate;

37  import java.util.List;
38  
39  import org.hibernate.Criteria;
40  import org.hibernate.criterion.Order;
41  import org.hibernate.criterion.Restrictions;
42  import org.springframework.stereotype.Repository;
43  
44  import fr.mcc.ginco.beans.NodeLabel;
45  import fr.mcc.ginco.beans.ThesaurusVersionHistory;
46  import fr.mcc.ginco.dao.IThesaurusVersionHistoryDAO;
47  
48  /**
49   * Implementation of the data access object to the thesaurus_version_history database table
50   *
51   */
52  @Repository("thesaurusVersionHistoryDAO")
53  public class ThesaurusVersionHistoryDAO extends
54  		GenericHibernateDAO<ThesaurusVersionHistory, String> implements IThesaurusVersionHistoryDAO {	
55  
56  	public ThesaurusVersionHistoryDAO() {
57  		super(ThesaurusVersionHistory.class);
58  	}
59  
60  	@Override
61  	public List<ThesaurusVersionHistory> findVersionsByThesaurusId(
62  			String thesaurusId) {
63  		Criteria criteria = getCurrentSession().createCriteria(
64  				ThesaurusVersionHistory.class, "tv");
65  		selectThesaurus(criteria, thesaurusId);
66  		descOrderBy(criteria, "date");
67  		return criteria.list();
68  	}
69  	
70  	@Override
71  	public List<ThesaurusVersionHistory> findAllOtherThisVersionTrueByThesaurusId(
72  			String thesaurusId, String excludedVersionId) {
73  		Criteria criteria = getCurrentSession().createCriteria(
74  				ThesaurusVersionHistory.class, "tv");
75  		selectThesaurus(criteria, thesaurusId);
76  		excludeAVersion(criteria, excludedVersionId);
77  		isThisVersion(criteria, true);
78  		return criteria.list();
79  	}
80  
81  	@Override
82  	public ThesaurusVersionHistory findThisVersionByThesaurusId(String thesaurusId) {
83  		Criteria criteria = getCurrentSession().createCriteria(
84  				ThesaurusVersionHistory.class, "tv");
85  		selectThesaurus(criteria, thesaurusId);
86  		isThisVersion(criteria, true);
87  		
88  		List<ThesaurusVersionHistory>foundVersions = criteria.list();
89  		if (foundVersions.size()>0) {
90  			return (ThesaurusVersionHistory) criteria.list().get(0);
91  		}
92  		return null;
93  	}
94  
95  	private void selectThesaurus(Criteria criteria, String thesaurusId) {
96          criteria.add(Restrictions.eq("tv.thesaurus.identifier", (String)thesaurusId));
97      }
98  	
99  	private void excludeAVersion(Criteria criteria, String excludedVersionId) {
100         criteria.add(Restrictions.ne("tv.identifier", (String)excludedVersionId));
101     }
102 	
103 	private void isThisVersion(Criteria criteria, Boolean isThisVersion) {
104         criteria.add(Restrictions.eq("tv.thisVersion", isThisVersion));
105     }
106 	
107 	private void descOrderBy(Criteria criteria, String sortColumn) {
108         criteria.addOrder(Order.desc(sortColumn));
109     }
