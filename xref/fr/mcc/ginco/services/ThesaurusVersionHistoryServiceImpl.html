56  @Service("thesaurusVersionHistoryService")
57  public class ThesaurusVersionHistoryServiceImpl implements IThesaurusVersionHistoryService {

59      @Inject
60      @Named("thesaurusVersionHistoryDAO")
61      private IThesaurusVersionHistoryDAO thesaurusVersionHistoryDAO;
62

64      @Named("generatorService")
65      private IIDGeneratorService generatorService;

68      private Logger logger;

71  	public List<ThesaurusVersionHistory> getVersionsByThesaurusId(
72  			String thesaurusId) {

74  	}
75  
76  	@Override
77  	public ThesaurusVersionHistory getThesaurusVersionHistoryById(String id) {
78  		return thesaurusVersionHistoryDAO.getById(id);
79  	}
80  
81      @Override
82      @Transactional(readOnly=false)
83      public ThesaurusVersionHistory publishThesaurus(Thesaurus thesaurus, String userId) {
84          ThesaurusVersionHistory newVersion = new ThesaurusVersionHistory();
85          newVersion.setThesaurus(thesaurus);
86          newVersion.setThisVersion(true);
87          newVersion.setDate(DateUtil.nowDate());
88          newVersion.setStatus(ThesaurusVersionStatusEnum.PUBLISHED.getStatus());
89          newVersion.setUserId(userId);
90          newVersion.setIdentifier(generatorService.generate(ThesaurusVersionHistory.class));
91          return createOrUpdateVersion(newVersion);
92      }
93  
94      @Override
95  	@Transactional(readOnly=false)
96  	public ThesaurusVersionHistory createOrUpdateVersion(ThesaurusVersionHistory version) {
97  		if (version.getThisVersion() == true) {
98  			//We set attribute thisVersion for all versions to false
99  			List<ThesaurusVersionHistory> allOtherVersions = new ArrayList<ThesaurusVersionHistory>();
100 			if (version.getThesaurus() != null) {
101 				allOtherVersions = thesaurusVersionHistoryDAO.findAllOtherThisVersionTrueByThesaurusId(version.getThesaurus().getIdentifier(), version.getIdentifier());				
102 			}
103 			for (ThesaurusVersionHistory thesaurusVersionHistory : allOtherVersions) {
104 				thesaurusVersionHistory.setThisVersion(false);
105 			}
106 		} else {
107 			//We verify there is at least one version to true for the flag thisVersion (thisVersion)
108 			if ( thesaurusVersionHistoryDAO.findThisVersionByThesaurusId(version.getThesaurus().getIdentifier()) == null) {
109 				throw new BusinessException("A version must be set as the current version", "no-current-version");
110 			}
111 		}
112 		return thesaurusVersionHistoryDAO.update(version);
113 	}
