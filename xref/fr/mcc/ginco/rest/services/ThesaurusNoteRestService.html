127 	/**
128 	 * Public method used to get the list of all notes for a concept or a term.
129 	 * 
130 	 * @return list of Note objects for a term, if not found - {@code null}
131 	 * @throws BusinessException 
132 	 */
133 	@GET
134 	@Path("/getNotes")
135 	@Produces({ MediaType.APPLICATION_JSON })
136 	public ExtJsonFormLoadData<List<ThesaurusNoteView>> getNotes(
137 			@QueryParam("conceptId") String conceptId,
138 			@QueryParam("termId") String termId,
139 			@QueryParam("start") Integer startIndex,
140 		     @QueryParam("limit") Integer limit) throws BusinessException {
141 		
142 		logger.info("Getting Notes for a concept or a term with following parameters : " + "conceptId " +conceptId + " and termId " + termId + " and startIndex " + startIndex + " with a limit of " + limit);
143 		
144 		List<Note> notes = new ArrayList<Note>();
145 		Long total;
146 		if (conceptId != null) {
147 			notes = noteService.getConceptNotePaginatedList(conceptId, startIndex, limit);
148 			total = noteService.getConceptNoteCount(conceptId);
149 		} else if (termId != null) {
150 			notes = noteService.getTermNotePaginatedList(termId, startIndex, limit);
151 			total = noteService.getTermNoteCount(termId);
152 		} else {
153 			throw new BusinessException("You need to specify an id for the concept or the term", "conceptid-or-termid-needed");
154 		}
155 		ExtJsonFormLoadData<List<ThesaurusNoteView>> result = new ExtJsonFormLoadData<List<ThesaurusNoteView>>(thesaurusNoteViewConverter.convert(notes));
156 		result.setTotal(total);
157 		return result;
158 	}

160 	/**
161 	 * Public method used to create new notes
162 	 * @throws BusinessException 
163 	 */
164 	@POST
165 	@Path("/createNotes")
166 	@Consumes({ MediaType.APPLICATION_JSON })
167 	@PreAuthorize("hasRole('ROLE_ADMIN')")
168 	public ExtJsonFormLoadData<List<ThesaurusNoteView>> createNotes(List<ThesaurusNoteView> noteViews, @QueryParam("conceptId") String conceptId, @QueryParam("termId") String termId) throws BusinessException {
169 		//We set for each note if it belongs to a concept or a term
170 		for (ThesaurusNoteView view : noteViews) {
171 			if(conceptId != null) {
172 				logger.info("Updating notes for conceptid : " + conceptId);
173 				view.setConceptId(conceptId);
174 			} else if (termId != null){
175 				logger.info("Updating notes for termid : " + termId);
176 				view.setTermId(termId);
177 			} else {
178 				throw new BusinessException("You need to specify an id for the concept or the term", "conceptid-or-termid-needed");
179 			}
180 		}
181 		
182 		List<Note> notes = thesaurusNoteViewConverter.convertToNote(noteViews);
183 		
184 		List<Note> resultNotes = new ArrayList<Note>() ;
185 		for (Note note : notes) {
186 			if (!note.getLexicalValue().isEmpty() && note.getLanguage() != null && note.getNoteType() != null) {
187 				resultNotes.add(noteService.createOrUpdateNote(note));
188 				indexerService.addNote(note);
189 			} else {
190 				throw new BusinessException("Failed to update the note : fields required not filled in", "note-update-failed-empty-fields");
191 			}
192 		}
193 
194 		//Returns all the created notes converted
195 		return new ExtJsonFormLoadData<List<ThesaurusNoteView>>(thesaurusNoteViewConverter.convert(resultNotes));
196 		
197 	}
