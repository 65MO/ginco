114 	@GET
115 	@Path("/getConcept")
116 	@Produces({ MediaType.APPLICATION_JSON })
117 	public ThesaurusConceptView getConceptById(
118 			@QueryParam("id") String conceptId) throws BusinessException {
119 
120         String resultId = "";
121 
122         if(conceptId.startsWith(ChildrenGenerator.ID_PREFIX)) {
123             resultId = conceptId.substring(
124                     conceptId.indexOf(ChildrenGenerator.PARENT_SEPARATOR)
125                             + ChildrenGenerator.PARENT_SEPARATOR.length());
126         } else {
127         	resultId = conceptId;
128         }
129 
130         List<ThesaurusTerm> terms = new ArrayList<ThesaurusTerm>();
131         logger.debug("Trying to load concept with id = " + resultId);
132         
133 		terms = thesaurusTermService.getTermsByConceptId(resultId);
134 		return thesaurusConceptViewConverter.convert(
135 				thesaurusConceptService.getThesaurusConceptById(resultId),
136 				terms);
137 	}

139 	/**
140 	 * Public method used to create or update a concept
141 	 * 
142 	 * @throws BusinessException
143 	 */
144 	@POST
145 	@Path("/updateConcept")
146 	@Consumes({ MediaType.APPLICATION_JSON })
147 	@PreAuthorize("hasRole('ROLE_ADMIN')")
148 	public ThesaurusConceptView updateConcept(
149 			ThesaurusConceptView thesaurusConceptViewJAXBElement)
150 			throws BusinessException, TechnicalException {
151 
152 		ThesaurusConcept convertedConcept = thesaurusConceptViewConverter
153 				.convert(thesaurusConceptViewJAXBElement);
154 
155 		List<ThesaurusTerm> terms = termViewConverter
156 				.convertTermViewsInTerms(thesaurusConceptViewJAXBElement
157                         .getTerms(), true);
158 		logger.info("Number of converted terms : " + terms.size());
159 
160 		List<ThesaurusTerm> preferedTerm = thesaurusTermService
161 				.getPreferedTerms(terms);
162 
163 		// Business rule : a concept must have at least 1 term
164 		if (preferedTerm.size() == 0) {
165 			throw new BusinessException("A concept must have a prefered term",
166 					"missing-preferred-term-for-concept");
167 		}
168 
169 		// Business rule : a concept mustn't have more than one prefered term
170 		if (preferedTerm.size() > 1) {
171 			throw new BusinessException(
172 					"A concept must have at only one prefered term",
173 					"to-many-preferred-terms-for-concept");
174 		}
175 
176 		if (StringUtils.isNotEmpty(thesaurusConceptViewJAXBElement.getIdentifier())) {
177 				List<ThesaurusTerm> origin = thesaurusTermService
178 						.getTermsByConceptId(convertedConcept.getIdentifier());
179 				thesaurusTermService.markTermsAsSandboxed(terms, origin);
180 			
181 		}
182 
183 		// We save or update the concept
184 		logger.info("Saving concept in DB");
185 
186 		ThesaurusConcept returnConcept = thesaurusConceptService
187 				.updateThesaurusConcept(convertedConcept, terms, thesaurusConceptViewJAXBElement.getAssociatedConcepts());
188 		
189 		for (ThesaurusTerm term : terms) {
190 			indexerService.addTerm(term);
191 		}
192 		indexerService.addConcept(convertedConcept);
193 		// Return ThesaurusConceptView created/updated
194 		return thesaurusConceptViewConverter.convert(returnConcept, terms);
195 	}

197     @GET
198     @Path("/getConcepts")
199     @Produces({ MediaType.APPLICATION_JSON })
200     public List<ThesaurusConceptReducedView> getConceptsByThesaurusId(
201             @QueryParam("id") String conceptId,
202             @QueryParam("thesaurusId") String thesaurusId,
203             @QueryParam("searchOrphans") @Nullable String searchOrphans,
204             @QueryParam("onlyValidatedConcepts") @Nullable Boolean onlyValidatedConcepts)
205             throws BusinessException {
206 
207         Boolean searchOrphanParam;
208 
209         if(searchOrphans == null) {
210             searchOrphanParam = null;
211         } else if (searchOrphans.isEmpty()) {
212             searchOrphanParam = null;
213         } else {
214             searchOrphanParam = Boolean.parseBoolean(searchOrphans);
215         }
216         
217         boolean onlyValidated = onlyValidatedConcepts;
218         if (onlyValidatedConcepts == null) {
219         	//By default, if the value is not set, we return all concepts
220         	onlyValidated = false;
221         }
222 
223         return thesaurusConceptViewConverter
224                 .convert(thesaurusConceptService.getConceptsByThesaurusId(conceptId, thesaurusId,
225                         searchOrphanParam, onlyValidated));
226     }

290 	 * Public method to get all status for concept (id + label)
291 	 * The types are read from a properties file
292 	 * @throws BusinessException 
293 	 */
294 	@GET
295 	@Path("/getAllConceptStatus")
296 	@Produces({MediaType.APPLICATION_JSON})
297 	public ExtJsonFormLoadData<List<GenericStatusView>> getAllConceptStatus() throws BusinessException {
298 		List<GenericStatusView> listOfStatus = new ArrayList<GenericStatusView>();
299 		
300 		try {
301 			ResourceBundle res = ResourceBundle.getBundle("labels", new EncodedControl("UTF-8"));
302 			String availableStatusIds[] = res.getString("concept-status").split(",");
303 			
304 			if ("".equals(availableStatusIds[0])) {
305 				//Ids of status for concepts are not set correctly
306 				throw new BusinessException("Error with property file - check values of identifier concept status", "check-values-of-concept-status");
307 			}
308 			
309 	        for (String id : availableStatusIds) {
310 	        	GenericStatusView conceptStatusView = new GenericStatusView();
311 	        	conceptStatusView.setStatus(Integer.valueOf(id));
312 	        	
313 	        	String label = res.getString("concept-status["+ id +"]");
314 	        	if (label.isEmpty()) {
315 	        		//Labels of status are not set correctly
316 	        		throw new BusinessException("Error with property file - check values of identifier concept status", "check-values-of-concept-status");
317 				} else {
318 					conceptStatusView.setStatusLabel(label);
319 				}
320 	        	listOfStatus.add(conceptStatusView);
321 			}
322 		} catch (MissingResourceException e) {
323 			throw new BusinessException("Error with property file - check values of concept status", "check-values-of-concept-status", e);
324 		}
325 		ExtJsonFormLoadData<List<GenericStatusView>> result = new ExtJsonFormLoadData<List<GenericStatusView>>(listOfStatus);
326         result.setTotal((long) listOfStatus.size());
327 		return result;
328 	}
