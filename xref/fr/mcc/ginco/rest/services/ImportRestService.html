62  /**
63   * Base REST service intended to be used for SKOS Import the @Produces({
64   * MediaType.TEXT_HTML}) is no mistake : this rest service is used by an ajax
65   * call and IE cannot display result if JSOn is returned
66   */
67  @Service
68  @Path("/importservice")
69  @Produces({ MediaType.TEXT_HTML })
70  @PreAuthorize("hasRole('ROLE_ADMIN')")
71  public class ImportRestService {
72  	@Inject
73      @Context
74  	private javax.servlet.ServletContext servletContext;
75  
76  	@Inject
77  	@Named("skosImportService")
78  	private ISKOSImportService skosImportService;
79  
80  	@Inject
81  	@Named("thesaurusViewConverter")
82  	private ThesaurusViewConverter thesaurusViewConverter;	
83  
84  	/**
85  	 * This method is called to import a SKOS thesaurus the @Produces({
86  	 * MediaType.TEXT_HTML}) is no mistake : this rest service is used by an
87  	 * ajax call and IE cannot display result if JSOn is returned
88  	 * 
89  	 * @param body
90  	 * @param request
91  	 * @return The imported thesaurus in JSOn string representig a
92  	 *         ExtJsonFormLoadData
93  	 * @throws BusinessException
94  	 * @throws JsonGenerationException
95  	 * @throws JsonMappingException
96  	 * @throws IOException
97  	 */
98  	@POST
99  	@Path("/import")
100 	@Consumes(MediaType.MULTIPART_FORM_DATA)
101 	@Produces(MediaType.TEXT_HTML)
102 	public String uploadFile(MultipartBody body,
103 			@Context HttpServletRequest request) throws BusinessException,
104 			JsonGenerationException, JsonMappingException, IOException {
105 		Attachment file = body.getAttachment("import-file-path");
106 		String content = file.getObject(String.class);
107 		String fileName = file.getDataHandler().getName();
108 		File tempdir = (File) servletContext
109 				.getAttribute("javax.servlet.context.tempdir");
110 
111 		Thesaurus importedThesaurus = skosImportService.importSKOSFile(content,
112 				fileName, tempdir);
113 		ObjectMapper mapper = new ObjectMapper();
114 		String serialized = mapper.writeValueAsString(new ExtJsonFormLoadData(
115 				thesaurusViewConverter.convert(importedThesaurus)));
116 		return serialized;
117 
118 	}
